Implementar handler:

Arquivo: `event_handlers.py`

```python
import json
import time
import redis.asyncio as aioredis

TRADE_BUFFER_TTL = 90

async def handle_spot_trades(result: list[dict]) -> None:
    r = aioredis.from_url(REDIS_URL)
    now_ms = int(time.time() * 1000)

    for trade in result:
        symbol = trade.get("currency_pair")
        if not symbol:
            continue

        key = f"trades_buffer:{symbol}"

        await r.zadd(key, {json.dumps(trade): trade.get("create_time_ms", now_ms)})
        await r.zremrangebyscore(key, 0, now_ms - TRADE_BUFFER_TTL * 1000)
        await r.expire(key, TRADE_BUFFER_TTL * 2)
```

Registrar handler:

```python
client.register_handler("spot.trades", handle_spot_trades)
```
