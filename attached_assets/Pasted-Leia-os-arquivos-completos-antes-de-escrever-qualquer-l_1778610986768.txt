4b. Adicionar função antes de _scan_async (ou equivalente):

  async def _inject_live_order_flow(
      symbol, indicators, pool_id, user_id,
      config_service, order_flow_service, logger,
  ):
      try:
          cfg = await config_service.get_config(
              user_id=user_id, pool_id=pool_id, config_type="pipeline"
          )
          window  = int(cfg.get(_ORDER_FLOW_WINDOW_CONFIG_KEY,  _ORDER_FLOW_WINDOW_DEFAULT))
          max_age = int(cfg.get(_ORDER_FLOW_MAX_AGE_CONFIG_KEY, _ORDER_FLOW_MAX_AGE_DEFAULT))

          live = await order_flow_service.get_order_flow_data(
              symbol=symbol, window_seconds=window
          )
          if live is None:
              logger.warning(f"[L3][{symbol}] order_flow buffer vazio — fallback")
              return indicators, True

          age = live.get("data_age_seconds")
          if age is not None and max_age > 0 and age > max_age:
              logger.warning(f"[L3][{symbol}] STALE BLOCK age={age}s > max={max_age}s")
              return indicators, False

          updated = dict(indicators)
          for k, f in _LIVE_ORDER_FLOW_FIELDS.items():
              if live.get(f) is not None:
                  updated[k] = live[f]

          logger.debug(f"[L3][{symbol}] order flow injetado age={age}s window={window}s")
          return updated, True

      except Exception as exc:
          logger.error(f"[L3][{symbol}] _inject_live_order_flow erro: {exc!r} — fallback")
          return indicators, True
