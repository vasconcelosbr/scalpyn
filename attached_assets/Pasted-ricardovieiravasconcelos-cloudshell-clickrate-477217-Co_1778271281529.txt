SERVICES=(scalpyn scalpyn-worker-micro scalpyn-worker-structural scalpyn-worker-execution scalpyn-beat)
echo "==> revisão atual + Ready de cada service"
for svc in "${SERVICES[@]}"; do
  REV=$(gcloud run services describe "$svc" --region=us-central1 --format='value(status.latestReadyRevisionName)')
  IMG=$(gcloud run services describe "$svc" --region=us-central1 --format='value(spec.template.spec.containers[0].image)' | sed 's|.*/||')
  READY=$(gcloud run services describe "$svc" --region=us-central1 \
    --flatten='status.conditions[]' \
    --filter='status.conditions.type=Ready' \
    --format='value(status.conditions.status)')
  printf "  %-30s rev=%-30s ready=%-8s img=%s\n" "$svc" "$REV" "$READY" "$IMG"
done
