      _EXIT_FLOW_WINDOW_CONFIG_KEY   = "monitor_exit_flow_window_seconds"

      _EXIT_FLOW_MAX_AGE_CONFIG_KEY  = "monitor_exit_flow_max_age_seconds"

      _EXIT_TAKER_BEAR_DEFAULT       = 0.35  # taker_ratio < 0.35 = pressão vendedora

  3b. Adicionar função _check_exit_signals():

      async def _check_exit_signals(
          self,
          position,           # objeto posição com symbol, entry_price, side
          current_price: float,
          config_service,
          order_flow_service,
      ) -> tuple[bool, str]:
          """
          Avalia sinais de saída dinâmica baseados em order flow ao vivo.
          Retorna (deve_sair, motivo).
          Nunca bloqueia saída por TP/SL — apenas adiciona condições de saída antecipada.
          Falha silenciosa: se order_flow indisponível, retorna (False, "flow_unavailable").
          """
          try:
              # Ler config (Zero Hardcode)
              cfg = await config_service.get_config(
                  user_id=position.user_id,
                  pool_id=position.pool_id,
                  config_type="trade_monitor",
              )
              window  = int(cfg.get(_EXIT_FLOW_WINDOW_CONFIG_KEY, _EXIT_FLOW_WINDOW_DEFAULT))
              max_age = int(cfg.get(_EXIT_FLOW_MAX_AGE_CONFIG_KEY, _EXIT_FLOW_MAX_AGE_DEFAULT))
              bear_threshold  = float(cfg.get(_EXIT_TAKER_BEAR_CONFIG_KEY, _EXIT_TAKER_BEAR_DEFAULT))
              spike_threshold = float(cfg.get(_EXIT_VOLUME_SPIKE_CONFIG_KEY, _EXIT_VOLUME_SPIKE_DEFAULT))

              flow = await order_flow_service.get_order_flow_data(
                  symbol=position.symbol,
                  window_seconds=window,
              )

              if flow is None:
                  return False, "flow_unavailable"

              age = flow.get("data_age_seconds")
              if age is not None and age > max_age:
                  return False, f"flow_stale_{age:.0f}s"

              taker_ratio  = flow.get("taker_ratio", 0.5)
              volume_delta = flow.get("volume_delta", 0.0)

              # Sinal 1: pressão vendedora dominante
              # taker_ratio < threshold indica que vendedores agressivos dominam
              if taker_ratio < bear_threshold:
                  return True, f"taker_bear_{taker_ratio:.3f}"

              # Sinal 2: exaustão de volume (delta negativo expressivo)
              # volume_delta muito negativo após entrada = pump acabou
              taker_buy  = flow.get("taker_buy_volume", 0.0)
              taker_sell = flow.get("taker_sell_volume", 0.0)
              total_vol  = taker_buy + taker_sell
              if total_vol > 0 and volume_delta < 0:
                  sell_dominance = abs(volume_delta) / total_vol
                  if sell_dominance > (1 - (1 / spike_threshold)):
                      return True, f"volume_exhaustion_{sell_dominance:.2f}"

              return False, "flow_ok"

          except Exception as exc:
              logger.error(
                  f"[Monitor][{position.symbol}] _check_exit_signals erro: {exc!r} — ignorando"
              )
              return False, "flow_error"
