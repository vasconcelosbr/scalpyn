Crie migration Alembic para nova tabela trade_decisions:

  id              UUID PRIMARY KEY DEFAULT gen_random_uuid()
  trace_id        VARCHAR(64) NOT NULL
  user_id         UUID REFERENCES users(id)
  pool_id         UUID REFERENCES pools(id)
  symbol          VARCHAR(20) NOT NULL
  market_type     VARCHAR(10) NOT NULL  -- spot | futures
  exchange        VARCHAR(50)
  decided_at      TIMESTAMPTZ DEFAULT now()
  status          VARCHAR(20) NOT NULL  -- APPROVED | REJECTED | BLOCKED | SKIPPED
  stage           VARCHAR(10) NOT NULL  -- L1 | L2 | L3 | EXECUTION
  reason          TEXT                  -- texto humano da reprovação
  blocking_rule   VARCHAR(255)          -- nome exato da regra que bloqueou
  rule_details    JSONB                 -- {"rule": "rsi", "value": 74.2, "threshold": "< 68", "operator": "<", "approved": false}
  rules_matched   JSONB                 -- lista de regras aprovadas
  rules_failed    JSONB                 -- lista de regras reprovadas com detalhes
  rules_skipped   JSONB                 -- lista de regras puladas
  score_breakdown JSONB                 -- scores L1/L2/L3 individuais
  indicators_snapshot JSONB            -- snapshot de todos os indicadores no momento da decisão
  latency_ms      JSONB                 -- {"l1": 12, "l2": 8, "l3": 45, "total": 65}
  trade_id        UUID REFERENCES trades(id) NULL  -- preenchido se APPROVED e trade criado

Índices:
  (user_id, decided_at DESC)
  (symbol, decided_at DESC)
  (status, decided_at DESC)
  (trace_id)
