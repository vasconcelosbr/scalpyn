PASSO 6 — Parabolic SAR

  def _calc_parabolic_sar(
      self,
      highs: list[float],
      lows: list[float],
      closes: list[float],

      af_increment: float = 0.02,
      af_max: float = 0.20,
  ) -> dict | None:
      """
      Parabolic SAR — algoritmo original de J. Welles Wilder Jr.

      FÓRMULAS:
        Em tendência de alta:
          SAR(n) = SAR(n-1) + AF × (EP - SAR(n-1))
          EP = maior máxima da tendência atual
          AF aumenta af_increment a cada novo EP, até af_max

        Em tendência de baixa:
          SAR(n) = SAR(n-1) - AF × (SAR(n-1) - EP)
          EP = menor mínima da tendência atual
          AF aumenta af_increment a cada novo EP, até af_max

      LÓGICA DE REVERSÃO:
        Bullish→Bearish: preço fecha abaixo do SAR calculado
          → SAR reinicia no EP da tendência anterior (maior máxima)
          → EP reinicia na mínima do candle atual
          → AF reinicia para af_start
        Bearish→Bullish: preço fecha acima do SAR calculado
          → SAR reinicia no EP da tendência anterior (menor mínima)
          → EP reinicia na máxima do candle atual
          → AF reinicia para af_start

      PROTEÇÃO ANTI-REPAINT:
        Em alta: SAR não pode ser maior que as mínimas dos 2 candles anteriores
          SAR = min(SAR_calc, low[n-1], low[n-2])
        Em baixa: SAR não pode ser menor que as máximas dos 2 candles anteriores
          SAR = max(SAR_calc, high[n-1], high[n-2])

      SEGURANÇA MATEMÁTICA:
        - Validar len(highs) == len(lows) == len(closes) >= 3
        - Evitar divisão por zero: verificar EP != SAR antes de calcular AF
        - Retornar None se candles insuficientes

      Retorna dict com:
        {
          "sar":           float,   # valor SAR atual
          "trend":         str,     # "bullish" ou "bearish"
          "ep":            float,   # Extreme Point atual
          "af":            float,   # Acceleration Factor atual
          "distance_pct":  float,   # abs(close - sar) / close * 100
          "signal":        str,     # "BUY", "SELL" ou "HOLD"
          "reversal":      bool,    # True se houve reversão neste candle
        }
      """

    sar_value, sar_trend, sar_ep, sar_af, sar_distance_pct, sar_signal, sar_reversal

  Parâmetros af_start, af_increment, af_max: ler do config_profiles via

    sar_af_start, sar_af_increment, sar_af_max
  Defaults: 0.02, 0.02, 0.20
